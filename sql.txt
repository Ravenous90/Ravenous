CREATE DATABASE users CHARACTER SET utf8 COLLATE utf8_general_ci;

CREATE TABLE `users`.`users` ( `id` INT NOT NULL AUTO_INCREMENT , 
`username` VARCHAR(16) NOT NULL , 
`email` VARCHAR(32) NOT NULL , 
`password` VARCHAR(16) NOT NULL , 
`gender` BOOLEAN NULL DEFAULT NULL , 
`phone` VARCHAR(16) NULL DEFAULT NULL , 
`age` TINYINT UNSIGNED NULL DEFAULT NULL , 
`info` VARCHAR(128) NULL DEFAULT NULL , 
`verification` TINYINT(1) NOT NULL DEFAULT '0', PRIMARY KEY (`id`)) ENGINE = InnoDB;
